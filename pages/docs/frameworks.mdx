import { Card, Cards } from "nextra/components";
import cmdArr from "./frameworks/_meta.json";

# Frameworks

Use the following buttons to navigate throught content

export const FrmLinks = () => {
  var cmdObj = JSON.parse(JSON.stringify(cmdArr));
  // console.log(typeof cmdObj, cmdObj);
  return (
    <Cards>
      {Object.keys(cmdObj).map((key) => (
        <Card
          key={key}
          title={`${cmdObj[key]}`}
          href={`/docs/frameworks/${key}`}
        />
      ))}
    </Cards>
  );
};

<FrmLinks />
